111 inline std::vector<int>
112 jet_indices_comb(int n_jets ,
113  int n_to_choose ) {
114  // returns the indices of the jets to choose, according to combinatorics
115  namespace view = ranges::view;
116  namespace action = ranges::action;
117  // hardcode default seed 5 for reproducability
118  static std::seed_seq seeds({5});
119  static std::mt19937_64 engine(seeds);
120 
121  std::binomial_distribution<> binom(n_to_choose, f);
122  int n_to_return = 0;
123  do {
124  n_to_return = binom(engine);
125  } while (n_to_return < n_to_choose);
126 
127  return (std::vector<int>(view::ints(0, n_jets)) | action::shuffle(engine)
128  | action::take(n_to_return));
129 }
